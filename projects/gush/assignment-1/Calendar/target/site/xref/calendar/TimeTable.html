24      /**
25       * Retrieves a range of appointments between two dates.
26       * @return A list of all of the CalDays between firstDate (inclusive)
27       *  and lastDate (exclusive) with their respective appointments. 
28       * @throws DateOutOfRangeException If any of the days constructed by the
29       *  given values are invalid, or if lastDay is not after firstDay.
30       **/
31  	  public LinkedList<CalDay> getApptRange(LinkedList<Appt> appts,GregorianCalendar firstDay, GregorianCalendar lastDay)throws DateOutOfRangeException{
32  		  
33  		     //Create a linked list of calendar days <CalDay> to return
34  	        LinkedList<CalDay> calDays = new LinkedList<CalDay>();
35  	     
36  
37  	        
38  	        //Make sure that the first day is before the last day
39  	        if (!firstDay.before(lastDay)) {
40  	        	throw new DateOutOfRangeException ("Second date specified is not  before the first date specified.");
41  	        }
42  	        
43  	        
44  	        //Create the first CalDay object with the starting date and add to list
45  	        GregorianCalendar nextDay = (GregorianCalendar) firstDay.clone();
46  	        while (nextDay.before(lastDay)) {
47  
48  	            calDays.add(new CalDay(nextDay));
49  	            nextDay.add(nextDay.DAY_OF_MONTH, 1);
50  	        }
51  	        
52  	        //Retrieve the appts - <appt> 
53  		for (int i = 0; i < appts.size(); i++) {
54  			Appt appt=appts.get(i);
55  			if(!appt.getValid()) continue;
56  			// Figure out which days the appointment occurs on
57  			LinkedList<GregorianCalendar> apptOccursOnDays = getApptOccurences(
58  					appt, firstDay, lastDay);
59  
60  			// For each day in the list, calculate the difference between the
61  			// first day and the day of occurrence and add the appointment to
62  			// the correct CalDay
63  			int daysDifference = 0;
64  			nextDay = (GregorianCalendar) firstDay.clone();
65  			Iterator<GregorianCalendar> itr = apptOccursOnDays.iterator();
66  			while (itr.hasNext()) {
67  				GregorianCalendar apptOccursOn = (GregorianCalendar) itr.next();
68  
69  				while (nextDay.before(apptOccursOn)) {
70  					daysDifference++;
71  					nextDay.add(nextDay.DAY_OF_MONTH, 1);
72  				}
73  
74  				CalDay calDayOfAppt = (CalDay) calDays.get(daysDifference);
75  				calDayOfAppt.addAppt(appt);
76  
77  			}
78  
79  		}
80  		  return calDays;
81  	  }
82  	   /**
83  	     * This takes the given appointment and constructs a linked list of 
84  	     * GregorianCalendar's, each of which represent a day when the appointment
85  	     * occurs. The days are guaranteed to be between firstDay (inclusive) and
86  	     * lastDay (exclusive). They are guaranteed to be in order.
87  	     **/
88  	    private static LinkedList<GregorianCalendar> getApptOccurences(Appt appt, 
89  	        GregorianCalendar firstDay, GregorianCalendar lastDay) {
90  	        
91  	        LinkedList<GregorianCalendar> result = new LinkedList<GregorianCalendar>();
92  	        
93  	        //Make sure that the firstDay is before the last day
94  	        if (!firstDay.before(lastDay)) {
95  	            return result;
96  	        }
97  	        
98  	        //Get the first recurrence taken care of
99  	        GregorianCalendar occurrenceDay = 
100 	                new GregorianCalendar(appt.getStartYear(), appt.getStartMonth(), 
101 	                    appt.getStartDay());
102 	        
103 	        //If the first occurrence is after the last day, then it doesn't matter
104 	        //when it recurs because those dates must be after the last day too
105 	        if (!occurrenceDay.before(lastDay)) {
106 	            return result;
107 	        }
108 	        
109 	            
110 
111 	            //Make sure that there is a limited number of recurrences
112 	            for (int i = 0; i < appt.getRecurNumber()+1; i++) {
113 	                
114 	                //Add the day of occurrence to the list if it is after the first day
115 	                if (!occurrenceDay.before(firstDay)) {
116 	                    result.add(occurrenceDay);
117 	                }
118 	                
119 	                //Calculate the next recurrence day
120 	                occurrenceDay = getNextApptOccurrence(appt, occurrenceDay);
121 	                if (occurrenceDay == null) {
122 	                    break;
123 	                }
124 	                            
125 	                //Keep cycling while the occurence day is in range
126 	                if (!occurrenceDay.before(lastDay)) {
127 	                    break;
128 	                }
129 	            }        
130 	        return result;
131 	    }
132 	    /**
133 	     * Calculates the next recurring day in the given appointment. If the 
134 	     * appointment does not recur it returns null. If the date cannot be 
135 	     * calculated for some reason it returns null.
136 	     **/
137 	    private static GregorianCalendar getNextApptOccurrence(Appt appt, 
138 	            GregorianCalendar day) {
139 	        //If the appointment does not recur then return null
140 	        if (!appt.isRecurring()) {
141 	            return null;
142 	        }	        
143 	        //Leave the original day untouched.
144 	        GregorianCalendar nextDay = (GregorianCalendar)day.clone();
145 	        
146 	        //This depends on the recurrence settings
147 	        switch (appt.getRecurBy()) {
148 	            case Appt.RECUR_BY_WEEKLY:
149 	                int[] recurDays = appt.getRecurDays();
150 	                
151 	                //If the user specified weekly recurrence and didn't specify
152 	                //which week days, then assume it is the same week day of the
153 	                //first occurrence
154 	                if (recurDays.length == 0) {
155 	                    //Add 7 days and return that by default
156 	                    nextDay.add(nextDay.DAY_OF_MONTH, 7);
157 	                    return nextDay;
158 	                }
159 	                
160 	                //The user did specify weekly recurrence, so increment the
161 	                //day until it falls on a weekday the user specified
162 	                for (int k = 0; k < 7; k++) {
163 	                    nextDay.add(nextDay.DAY_OF_MONTH, 1);
164 	                    int newDayOfWeek = nextDay.get(nextDay.DAY_OF_WEEK);
165 	                
166 	                    for (int i = 0; i < recurDays.length; i++) {
167 	                        //If the calendar is set to a day of the week that the
168 	                        //appt recurs on then return that day.
169 	                        if (recurDays[i] == newDayOfWeek) {
170 	                            return nextDay;
171 	                        }
172 	                    }
173 	                }
174 	                
175 	                //The loop above should have found a day. If the program 
176 	                //execution is here then the weekdays specified are not in the
177 	                //range of valid Gregorian Calendar Days. Return null here.
178 	                return null;
179 	            case Appt.RECUR_BY_MONTHLY:
180 	                //Just increment the month and return the day. Not sure what
181 	                //happens when the day is 31 and the next month has 30 days...
182 	                nextDay.add(nextDay.MONTH, 1);
183 	                return nextDay;
184 	            case Appt.RECUR_BY_YEARLY:
185 	                //Just increment the year. The only possible problem is an 
186 	                //appointment that recurs on February 29.
187 	                nextDay.add(nextDay.YEAR, 1);
188 	                return nextDay;
189 	        }
190 	  
191 	        return null;
192 	    }

195 	    /**
196 	     * Deletes the appointment's information from the appointments data list. 
197 	     * @return updated list of appointments if the appointment is deleted successfully, otherwise null.
198 	     **/
199 	    public LinkedList<Appt> deleteAppt(LinkedList<Appt> appts,Appt appt) {
200 	    	//Do not do anything to appts equals to null 
201 	        if(appts==null||appt==null)
202         		return null;
203 	    	//Do not do anything to invalid appointments
204 	        if (!appt.getValid()) {
205 	            return null;
206 	        }
207 
208 	        //Remove the appointment from the list appts if applicable
209 	        
210 	        for(int i=1;i<appts.size()-1;i++){
211 	        	Appt tempAppt=appts.get(i);
212 	        	if(tempAppt.equals(appt)){
213 	        		appts.remove(i);
214 	        		return appts;
215 	        	}
216 	        		
217 	        }
218 	        return null;
219 	    }

221 	    public LinkedList<Appt> permute(LinkedList<Appt> appts, int[] pv){
222 	    	/* pv is an array of integers that specify a permutation (a
223 	    	   rearrangement or shuffling) of the appointments
224 	    	   
225 	    	   If the ith element of pv is k, then the ith element of 
226 	    	   appointments should be placed in the kth position.
227 	    	*/
228 	    	LinkedList<Appt> apptsUpdatedList=new LinkedList<Appt>(appts);   
229 	    	if(pv.length != appts.size())
230 	    	    throw new IllegalArgumentException();
231 
232 	    	int nexti =  0;
233 	    	for(int i = 0;i<pv.length;i++){
234 	    	    int newi = pv[nexti];
235 	    	    newi = pv[nexti];
236 	    	   Collections.swap(apptsUpdatedList,newi,newi);
237 	    	   nexti = newi;
238 	    	} 
239      		return apptsUpdatedList;
