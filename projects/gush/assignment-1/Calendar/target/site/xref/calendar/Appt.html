74       * Constructs a new appointment that starts at a specific time on the 
75       * date specified. The appointment is constructed with no recurrence 
76       * information by default. To set recurrence information, construct the
77       * appointment and then call setRecurrence(...) method.

81       * @param startDay The day of the month the appointment starts on.
82       * @param startMonth The month of the year the appointment starts on.
83       * @param startYear The year the appointment starts on.
84       * @param title The title or caption to give the appointment
85       * @param description The appointment's details
86       */
87      public Appt(int startHour, int startMinute, 
88              int startDay, int startMonth, int startYear,String title, String description)
89      {
90          //Sets all instance variables 
91      	this.startHour = startHour;
92      	this.startMinute = startMinute; 
93      	this.startDay = startDay; 
94      	this.startMonth = startMonth;
95      	this.startYear = startYear; 
96          setTitle(title);
97          setDescription(description);
98     
99          //Set default recurring information
100         int[] recurringDays = new int[0];
101         setRecurrence(recurringDays, RECUR_BY_MONTHLY, 0, RECUR_NUMBER_NEVER);
102 
103         
104         isValid();
105     }

109     /**
110      * @sets valid to true if the appointment is valid
111      */
112     private void isValid() {
113     	int NumDaysInMonth= CalendarUtil.NumDaysInMonth(startYear,startMonth-1);
114     				
115     	if(startHour<0 || startHour>23)
116     		this.valid=false;
117     	else
118         	if(startMinute<0 || startMinute>59)
119         		this.valid=false;
120         	else
121             	if(startDay<1 || startDay>NumDaysInMonth)
122             		this.valid=false;
123             	else
124                 	if(startMonth<1 || startMonth>12)
125                 		this.valid=false;
126                 	else
127                 		this.valid=true;
128     }

271     /**
272      * Generate a string representation for this appointment, with the
273      * form "11am: dentist" or "2:00pm: class".  The string consists of the
274      * 12-hour time representation with a (lower case) "am" or "pm"
275      * designator, followed by a colon and space, and then the appointment
276      * description.
277      * @return a printable representation of this appointment
278      */
279     private String represntationApp(){
280         String half = (getStartHour() > 11) ? "pm" : "am";
281         int printableHour = getStartHour();
282         if (printableHour > 11)
283         {
284             printableHour -= 12;
285         }
286         if (printableHour == 0)
287         {
288             printableHour = 12;
289         }
290         String represntationApp= printableHour +":"+ getStartMinute() + half;
291         return represntationApp;
292     	
293     }
294     public String toString()
295     {
296 		if (!getValid()) {
297 		    return null;
298 		}
299          String day= this.getStartMonth()+"/"+this.getStartDay()+"/"+this.getStartYear() + " at ";
300         return "\t"+ day +  this.represntationApp()  + " ," +  getTitle()+ ", "+  getDescription()+"\n";
301     }

303  //   The compareTo() method is hard to explain, in integer sorting, just remember
304  //   startMinute+startHour+day+month+year is ascending order.
305 	public int compareTo(Appt compareAppt) {
306 		int startMinute=	this.startMinute - ((Appt) compareAppt).getStartMinute();
307 		int startHour=	this.startHour - ((Appt) compareAppt).getStartHour();
308 		int day = this.getStartDay()-((Appt) compareAppt).getStartDay();
309 		int month = this.startMonth -((Appt) compareAppt).getStartMonth();
310 		int year = this.startYear -((Appt) compareAppt).getStartYear();
311 
312 
313 		//ascending order
314 
315 		return startMinute+startHour+day+month+year;
316 
317 	}
