14  /**
15   * Stores all of the appointments of a single calendar day. It also
16   * has some useful calendar-related abilities, such as the ability
17   * to create a new calendar day that is incremented by a day.
18   * 
19   * 
20   */
21  public class CalDay {
22  
23  	/** Collection of all of the appointments for the calendar day */
24  	LinkedList<Appt> appts;
25  	
26  	/** Stores whether or not this is a valid calendar day */
27  	boolean valid;
28  	
29  	/** Stores the calendar day */
30  	int day;
31  	
32  	/** Stores the calendar month */
33  	int month;
34  	
35  	/** Stores the calendar year */
36  	int year;
37  	
38  	/**
39  	* Default constructor
40  	* Constructs an invalid CalDay object
41  	*/
42  	public CalDay() {
43  		valid = false;
44  	}
45  	
46  	/**
47  	* Constructor
48  	* Creates a new CalDay which is ready for appointments to be
49  	* added to it. Note that months are numbered 0 through 11. This does
50  	* not check to see if a date is valid.
51  	*/
52  	public CalDay(GregorianCalendar cal) {
53  	
54  		int day = cal.get(cal.DAY_OF_MONTH);
55  		int month = cal.get(cal.MONTH);
56  		int year = cal.get(cal.YEAR);
57  	
58  		setDay(day);
59  		setMonth(month);
60  		setYear(year);
61  	
62  		setAppts(new LinkedList<Appt>());
63  	
64  		valid = true;
65  	}
66  	
67  	/**
68  	 * Adds an appointment to the calendar day object. The appointments are
69  	 * kept in order by start time. Note that this does not check to see if 
70  	 * the appointment actually occurs on this day. This is so the recurring
71  	 * appointments can be added. The appointment can also be added twice.
72  	 */
73  	public void addAppt(Appt appt) {
74  		if (appt.getValid()) {
75  			for (int i = 0; i < getAppts().size(); i++) {
76  				//Put the appointment in the correct order - finish this
77  				if (((Appt)getAppts().get(i)).getStartHour() >
78  										appt.getStartHour()) {
79  					
80  					getAppts().add(i, appt);
81  					return;
82  				}
83  			}
84  		    //The appointment hasn't been added yet, so add it
85  		    getAppts().add(appt);
86  		}
87  	}
88  
89  	
90  	/**
91  	 * @return True if this is an initalized CalDay object
92  	 */
93  	public boolean isValid() {
94  	    return valid;
95  	}
96  	
97  	/**
98  	 * This returns an iterator of Appt objects for the calendar day. The 
99  	 * appointments are added after the CalDay is constructed. The appointments
100 	 * are guaranteed to be in order by time, with appointments with no specific 
101 	 * time set placed at the beginning.
102 	 */
103 	public Iterator<?> iterator() {
104 	    if (isValid()) {
105 	        return getAppts().iterator();
106 	    }
107 	    else {
108 	        return null;
109 	    }
110 	}
111 	
112 	/** Sets appts */
113 	private void setAppts(LinkedList<Appt> appts) {
114 		if(appts!=null)
115 				this.appts = appts;
116 		
117 		if(appts!=null&&appts.size()==0)
118 			this.appts = appts;
119 	}
120 	
121 	/** Sets day */
122 	private void setDay(int day) {
123 	    this.day = day;
124 	}
125 	
126 	/** Sets month */
127 	private void setMonth(int month) {
128 	    this.month = month;
129 	}
130 	
131 	/** Sets year */
132 	private void setYear(int year) {
133 	    this.year = year;
134 	}
135 	
136 	/** Gets appts */
137 	public LinkedList<Appt> getAppts() {
138 	    return appts;
139 	}
140 	/** Gets size of the Appts */
141 	public int getSizeAppts() {
142 	    return appts.size();
143 	}		
144 	/** Gets day */
145 	public int getDay() {
146 	    return day;
147 	}
148 	
149 	/** 
150 	 * Gets the month represented by this calDay. Note that January-December
151 	 * are represented by 0-11 
152 	 */
153 	public int getMonth() {
154 	    return month;
155 	}
156 	
157 	/** Gets year */
158 	public int getYear() {
159 	    return year;
160 	}
161 
162 	/**
163 	 * Returns a string representation of the date represented in
164 	 * the format of MM/DD/YYYY. If someone where to ask what day of the
165 	 * year this CalDay was for, and they called this method, that would
166 	 * be the answer. No further adjustment would be necessary. Useful for
167 	 * debugging.
168 	 */
169 	public String toString() {
170 	     StringBuilder sb = new StringBuilder();
171 	
172 		if (isValid()) {
173 			String todayDate = (getMonth()) + "/" + getDay() + "/" + getYear();
174 			sb.append("\t --- " + todayDate + " --- \n");
175 			sb.append(" --- -------- Appointments ------------ --- \n");
176 			Iterator<Appt> itr = this.appts.iterator();
177 		    while(itr.hasNext()) {
178 		         Object element = itr.next();
179 		         
180 		         sb.append(element + " ");
181 		      }
182 		  
183 //			sb.append(this.appts);
184 			sb.append("\n");
185 		}
186        	 return sb.toString();
187 
188 	}
